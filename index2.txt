<script>
  emailjs.init("S8j3LoOQfZofPv41a"); // Public Key

  document.addEventListener("DOMContentLoaded", function() {
    // إضافة وإزالة المنتجات ديناميكيًا
    document.getElementById("addProduct").addEventListener("click", function() {
      const container = document.getElementById("products");
      const row = document.createElement("div");
      row.className = "product-row";
      row.innerHTML = `
        <input type="text" name="product_name[]" placeholder="اسم المنتج" required>
        <input type="number" name="product_quantity[]" placeholder="الكمية" required>
        <button type="button" class="remove">🗑️</button>
      `;
      container.appendChild(row);

      row.querySelector(".remove").addEventListener("click", function() {
        row.remove();
      });
    });

    // إرسال النموذج
    document.getElementById("rfqForm").addEventListener("submit", function(e) {
      e.preventDefault();
      const form = this;
      const msg = document.getElementById("message");

      // دمج كود الدولة مع رقم الهاتف
      const code = document.getElementById("countryCode").value;
      const phoneField = form.phone;
      phoneField.value = code + phoneField.value;

      // إنشاء جدول HTML للمنتجات
      const productNames = document.getElementsByName("product_name[]");
      const productQtys = document.getElementsByName("product_quantity[]");
      let table = "<table border='1' style='border-collapse:collapse;width:100%;text-align:center;'>";
      table += "<tr><th>المنتج</th><th>الكمية</th></tr>";
      for (let i = 0; i < productNames.length; i++) {
        table += `<tr><td>${productNames[i].value}</td><td>${productQtys[i].value}</td></tr>`;
      }
      table += "</table>";
      document.getElementById("products_table").value = table;

      emailjs.sendForm("service_j69z5hb", "template_oogionp", form)
        .then(function() {
          msg.className = "success";
          let counter = 5;
          msg.innerHTML = `✅ تم إرسال طلبك بنجاح! سيتم تحويلك خلال ${counter} ثوانٍ...`;

          const interval = setInterval(() => {
            counter--;
            msg.innerHTML = `✅ تم إرسال طلبك بنجاح! سيتم تحويلك خلال ${counter} ثوانٍ...`;
            if (counter === 0) {
              clearInterval(interval);
              window.location.href = "https://alalamia-electric.wuiltstore.com/";
            }
          }, 1000);
        }, function() {
          msg.className = "error";
          msg.innerHTML = "❌ فشل الإرسال! يرجى المحاولة مرة أخرى أو التواصل عبر: omaralalamiaa@gmail.com";
        });
    });
  });
</script>
